# Attributes


## Supported Types

### `boolean`

```typescript
const value: boolean = true;
```

### `string`

```typescript
const value: string = "<value>";
```

### `number`

```typescript
const value: number = 128403;
```

### `number`

```typescript
const value: number = 1284.03;
```

### `Date`

```typescript
const value: Date = new Date("2023-12-07T03:31:58.587Z");
```

### `string[]`

```typescript
const value: string[] = [
  "<value 1>",
  "<value 2>",
];
```

### `number[]`

```typescript
const value: number[] = [
  339048,
  133621,
  943799,
];
```

### `number[]`

```typescript
const value: number[] = [
  3390.48,
  1336.21,
  9437.99,
];
```

### `boolean[]`

```typescript
const value: boolean[] = [
  true,
];
```
